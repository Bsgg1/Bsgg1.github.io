A：排序后按从ram需求从小到大就可以了

#include<bits/stdc++.h>
using namespace std;
int t;
const int N=110;
int a[N],b[N];
struct node{
	int a,b;
	bool operator<(const node&m)
	{
		return a<m.a;
	}
}st[N];
int n,k;
int main()
{
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		for(int i=1;i<=n;i++) cin>>st[i].a;
		for(int i=1;i<=n;i++) cin>>st[i].b;
		sort(st+1,st+n+1);
		for(int i=1;i<=n;i++)
		{
			if(k<st[i].a) break;
			k+=st[i].b;
		}
		cout<<k<<"\n";
	}
	return 0;
}

B:偶数和奇数的gcd一定是1，所以把所有的奇数都变成偶数就可以了，特判只有一个奇数的情况

#include<bits/stdc++.h>
using namespace std;
int t;
const int N=110;
int a[N],b[N];
struct node{
	int a,b;
	bool operator<(const node&m)
	{
		return a<m.a;
	}
}st[N];
int n,k;
int main()
{
	cin>>t;
	while(t--)
	{
		int a,b,c;cin>>a>>b>>c;
		int k=b-a+1;
		if(b==a)
		{
		    if(a==1) puts("NO");
		    else puts("YES");
		}
		else{
		if(a%2==b%2&&a%2==1) k=k/2+1;
		else k=k/2;
		if(k<=c) puts("YES");
		else puts("NO");
		}
	}
	return 0;
}

C：每次找到最大的mex，按照题意模拟就可以

#include <bits/stdc++.h>
using namespace std;
int t;
#define LOCAL
const int N = 2e5 + 10;
int a[N];
int n, k;
int cnt[N];
int f[N];
int res[N];
int ccnt;
int ans[N];
int main()
{
#if ONLINE_JUDGE
#else
	freopen("in.txt", "r", stdin);
	freopen("out.txt", "w", stdout);
#endif
	scanf("%d", &t);
	while (t--)
	{
		ccnt = 0;
		scanf("%d", &n);
		for (int i = 0; i <= n; i++)
			res[i] = ans[i] = f[i] = cnt[i] = 0;
		for (int i = 1; i <= n; i++)
		{
			scanf("%d", &a[i]);
			res[a[i]]++;
		}
		int l = 0, r = 0;
		for (int i = 1; i <= n; i++)
		{
			f[a[i]]++;
			while (f[l])
				l++;
			if (!res[l])
			{
				ans[++ccnt] = l;
				for (int j = r + 1; j <= i; j++)
					f[a[j]] = 0, res[a[j]]--;
				r = i;
				l = 0;
			}
		}
		cout << ccnt << "\n";
		for (int i = 1; i <= ccnt; i++)
			cout << ans[i] << " ";
		puts("");
	}
	return 0;
}

D:如果存在回文串，只需要考虑3+2，2+3，以及reverse的情况就可以了

#include<bits/stdc++.h>
using namespace std;
map<string,vector<int>> mp;
set<string> se;
int t,n;
int main()
{
	cin>>t;
	while(t--)
	{
		mp.clear();
		se.clear();
		cin>>n;
		for(int i=1;i<=n;i++)
		{
			string s;cin>>s;
			se.insert(s);
			mp[s].push_back(i);
		}
		bool f=false;
		for(auto &[x,y]:mp)
		{
			string s=x;
			//cout<<x<<"\n";
			if(s.size()==3)
			{
				string s1="ab",s2="ab";
				s1[0]=s[2],s1[1]=s[1];
				s2[0]=s[1],s2[1]=s[0];
				if(se.count(s1)){
				if(*mp[s1].begin()<y.back()){
				f=true;
				break;
				}
				}
				if(se.count(s2))
				{
				if(*y.begin()<mp[s2].back()){
				f=true;
				break;
				}
				}
			}
			reverse(s.begin(),s.end());
			if(se.count(s))
			{
				f=true;
				break;
			}
		}
		if(f) puts("YES");
		else puts("NO");
	}
	return 0;
}

E:一个长为偶数的框，每次选一个点的四个邻点，一定可以通过选几次不重复地全部覆盖掉这个框，画画图就看明白了

#include<bits/stdc++.h>
using namespace std;
const int N=1100;
bool st[N][N];
int t,n;
int main()
{
	cin>>t;
	while(t--)
	{
		cin>>n;
		memset(st,false,sizeof st);
		int ans=0;
		for(int i=1;i<=n;i++)
		{
			for(int j=1;j<=n;j++)
			{
				int x;cin>>x;
				if(!st[i-1][j]&&!st[i+1][j]&&!st[i][j-1]&&!st[i][j+1])
				{
					ans^=x;
					st[i-1][j]=st[i+1][j]=st[i][j-1]=st[i][j+1]=true;
				}
			}
		}
		cout<<ans<<"\n";
	}


	return 0;
}
