D:考虑第一项为x，第i项是x+(i-1)*d，d是公差，然后很容易想到三分d，然后对于不同的d我们求得就是
x+(i-1)*d-ai，我们令数组ti表示ai-(i-1) d，然后就变成了经典的货仓选址问题，x等于ti的中位数就可以了

#include<bits/stdc++.h>
using namespace std;
#define in128 __int128
#define ll long long
#define LOCAL
int n;
const int N=2e5+10;
ll a[N];
ll c[N];
void print(in128 x)
{
	if(x>9) print(x/10);
	putchar(x%10+'0');
}
in128 check(ll d)
{
	for(int i=1;i<=n;i++)
	{
		c[i]=a[i]-d*(i-1);
	}
	sort(c+1,c+n+1);
	ll k=c[(n+1)/2];
	in128 res=0;
	for(int i=1;i<=n;i++)
	res+=abs(c[i]-k);
	return res;
}
int main()
{
	#if ONLINE_JUDGE
	#else
	freopen("in.txt","r",stdin);
	freopen("out.txt","w",stdout);
	#endif
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	in128 l=-1e13,r=1e13;
	while(l<r)
	{
		ll mid=l+(r-l)/3;
		ll midmid=r-(r-l)/3;
		if(check(mid)<=check(midmid)) r=midmid-1;
		else l=mid+1;
	}
	print(check(l));
	return 0;
}

C：如果先手拿的是当前剩余的里边的最大的，后手一定拿的也是最大的，所以最大的一定是成对出现的，然后其次就要奇偶，如果价值ci的有偶数个那么就是两人各一半，反之就是先手多1变成偶数的情况，然后我们考虑从小到大取，对于当前取到的价值，如果说他有k个，k是奇数就先手多拿一个然后变成偶数，现在双方各取k/2个，可以发现价值大的是可以穿插在价值小的中间拿的，但是价值最大的一定是成对出现的

#include<bits/stdc++.h>
using namespace std;
#define int long long
const int mod=998244353;
map<int,int> mp;
const int N=1e6+10;
int fac[N],infac[N];
int c[N];
int qpw(int a,int b)
{
    int res=1%mod;
    while(b)
    {
        if(b&1) res=res*a%mod;
        a=a*a%mod;
        b>>=1;
    }
    return res;
}
void init()
{
    fac[0]=1;
    for(int i=1;i<N;i++) fac[i]=fac[i-1]*i%mod;
    infac[N-1]=qpw(fac[N-1],mod-2);
    for(int i=N-2;i>=0;i--) infac[i]=infac[i+1]*(i+1)%mod;
}
int cal(int a,int b)
{
    if(a<b) return 0;
    return fac[a]*infac[b]%mod*infac[a-b]%mod;
}
signed main()
{
    int n;
    init();
    cin>>n;
    while(n--)
    {
        int x;cin>>x;
        c[x]++;
    }
    int sum=0,ans=1;
    for(int i=1;i<N;i++)
    {
        if(c[i]){
            int y=c[i];
        ans=ans*fac[y]%mod*cal(sum+(y)/2,(y)/2)%mod;
        sum+=y;
        }
    }
    cout<<ans<<"\n";

    return 0;
}
