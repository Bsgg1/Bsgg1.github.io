B:舔狗的故事
做法：首先是舔狗先选位置，女神会坐在距离他最远的地方，所以求出每个位置的距离，排个序输出就好了

#include <bits/stdc++.h>
#define x first
#define y second
#define LOCAL
using namespace std;
const int N = 550;
using pii = pair<int, int>;
pii que[N];

int t, n, k,m;
vector<int> e;
signed main()
{
#if ONLINE_JUDGE
#else
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
#endif
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        e.clear();
        for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
        {
    
            e.push_back(max(max(i-1+j-1,abs(i-n)+j-1),max(abs(i-n)+abs(j-m),i-1+abs(j-m))));

        }
        sort(e.begin(),e.end());
        for(auto x:e) cout<<x<<" ";
        puts("");
    }
    
    return 0;
}

c:
素数除去2剩下的都是奇数，若想相邻两条边加起来还是素数，那么最终的形式只能是一条链，然后边权23交替就好

#include<bits/stdc++.h>
#define LOCAL
#define x first
#define y second
using namespace std;
int t;
int n;
typedef pair<int,int> pii;
const int N=1e5+10;
int h[N],e[2*N],ne[2*N],idx;
pii q[N];
int c[N];
map<pii,int> mp;
int res;
void add(int a,int b)
{
    e[idx]=b,ne[idx]=h[a],h[a]=idx++;
}
void dfs(int u,int fa)
{
    for(int i=h[u];i!=-1;i=ne[i])
    {
        int j=e[i];
        if(j==fa) continue;
        mp[{u,j}]=mp[{j,u}]=res;
        res=5-res;
        dfs(j,u);
    }
}
int main()
{
    #if ONLINE_JUDGE
    #else
    freopen("in.txt","r",stdin);
    freopen("out.txt","w",stdout);
    #endif
    cin>>t;
    while(t--)
    {
        idx=0;
        mp.clear();
        memset(h,-1,sizeof h);
        cin>>n;
        for(int i=1;i<=n;i++) c[i]=0;
        bool fl=true;
        for(int i=1;i<n;i++)
        {
            int x,y;cin>>x>>y;
            add(x,y),add(y,x);
            q[i]={x,y};
            c[x]++,c[y]++;
        }
        int l;
        for(int i=1;i<=n;i++)
        {
            if(c[i]>2) 
            {
                fl=false;
                break;
            }
            if(c[i]==1) l=i;
        }
        //cout<<l<<"\n";
        res=2;
        if(!fl) puts("-1");
        else
        {
            dfs(l,-1);
            for(int i=1;i<n;i++) cout<<mp[q[i]]<<" ";
            puts("");
        }
    }


    return 0;
}

d:
由于gcd(a,b)小于等于ab的最小值，所以我们可以考虑对于一个数x若未出现在序列中，如果存在在序列中的大于x的倍数的gcd是x，那么通过一系列gcd操作就可以得到这个数

#include <bits/stdc++.h>
#define LOCAL
#define x first
#define y second
using namespace std;
int t;
int n;
typedef pair<int, int> pii;
const int N = 1e6 + 10;

bool st[N];

int main()
{
#if ONLINE_JUDGE
#else
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
#endif

    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        int x;
        cin >> x;
        st[x] = true;
    }
    int ans = 0;
    for (int i = 5e5; i; i--)
    {
        if (!st[i])
        {
            int k = 0;
            for (int j = i; j <= 1e6; j += i)
                if (st[j])
                    k = __gcd(k, j);
            if (k == i)
                ans++;
        }
    }
    cout << ans << "\n";

    return 0;
}
