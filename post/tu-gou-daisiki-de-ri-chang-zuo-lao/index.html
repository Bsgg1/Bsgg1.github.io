https://codeforces.com/contest/1508/problem/A
Tag:字符串，思维构造
题意：给三个由0和1构成的字符p_i，长度为2n，构造一个长度为3n的字符串S，让至少两个p是S的子序列
思路：首先可以得出任意p_i的cnt_0或者cnt_1是大于等于n的，考虑为什么是三个p?抽屉原理，一定会有两个p同时满足cnt1或者cnt0大于等于n，这样我们就固定好了n的长度，剩下的长度是2n,直接放就可以

#include <bits/stdc++.h>
using namespace std;
string s[3];
int f[3];
int t, n;
string check(int a, int b)
{
	char x;
	string ans;
	if (f[a] >= n)
		x = '1';
	else
		x = '0';
	int l = 0, r = 0;
	for (int i = 1; i <= n; i++)
	{
		while (l < 2 * n && s[a][l] != x)
			ans += s[a][l++];
		while (r < 2 * n && s[b][r] != x)
			ans += s[b][r++];
		l++, r++;
		ans += x;
	}
	while (l < 2 * n)
		ans += s[a][l++];
	while (r < 2 * n)
		ans += s[b][r++];
	return ans;
}
void solve()
{
	for (int i = 0; i < 3; i++)
	{
		for (int j = 0; j < 3; j++)
		{
			if (j == i)
				continue;
			if (f[i] > n && f[j] < n)
				continue;
			if (f[i] < n && f[j] > n)
				continue;
			string ss = check(i, j);
			if (ss.size() <= n * 3)
			{
				cout << ss << "\n";
				return;
			}
		}
	}
}
int main()
{
	cin >> t;
	while (t--)
	{
		cin >> n;
		f[0] = f[1] = f[2] = 0;
		for (int i = 0; i < 3; i++)
		{
			cin >> s[i];
			for (int j = 0; j < 2 * n; j++)
				if (s[i][j] == '1')
					f[i]++;
		}
		solve();
	}

	return 0;
}

https://codeforces.com/contest/1381/problem/A2
Tag：思维
字符串a，b，对a进行下列操作至多2n次使得a变成b，操作：选长度为k的前缀，其中0变成1，1变成0，并且翻转
思路：将a变为全0或者全1至多n次，然后考虑倒着匹配b，至多反转n次

#include <bits/stdc++.h>
using namespace std;
const int N=1e5+10;
vector<int> v;
char s1[N],s2[N];
int main()
{
    int t;
    scanf("%d",&t);
    while (t--)
    {
        int n;
        v.clear();
        scanf("%d%s%s",&n,s1+1,s2+1);
        if (n == 1)
           {
               if(s1[1]==s2[1]) puts("0");
               else puts("1 1");
           }
        else
        {
            char r=s1[n];
            for (int i = 2; i <=n; i++)
            {
                if (s1[i] != s1[i - 1])
                {
                    v.push_back(i-1);
                }
            }
            for(int i=n;i>=1;i--)
            {
                if(s2[i]!=r)
                {
                    v.push_back(i);
                    r='1'-r+'0';
                    //cout<<r<<"\n";
                }
            }
            cout<<v.size()<<" ";
            for(auto x:v) cout<<x<<" ";
            puts("");
        }
    }

    return 0;
}

https://codeforces.com/contest/1381/problem/B
Tag:思维转化后就是01背包
首先通过题意可以分析出连续减小的一段区间一定是属于一个数组的，然后就可以把连续下降的长度分别求出来，然后dp能不能得到n就好了

#include <bits/stdc++.h>
using namespace std;
const int N = 4e3 + 10;
int f[N], p[N], tot;
int dp[N];
int main()
{
    int t;
    cin >> t;
    while (t--)
    {
        int n;
        cin >> n;
        tot = 0;
        memset(dp, -1, sizeof dp);
        for (int i = 1; i <= 2 * n; i++)
            cin >> f[i];
        for (int l = 1, r; l <= 2 * n; l = r + 1)
        {
            r = l;
            while (r <= 2 * n && f[r] <= f[l])
                r++;
            r--;
            p[tot++] = r - l + 1;
            if (r == 2 * n)
                break;
        }
        memset(dp, -0x3f3f3f3f, sizeof(dp));
        dp[0] = 0;
        for (int i = 0; i < tot; i++)
        {
            for (int j = n; j >= p[i]; j--)
            {

                dp[j] = max(dp[j], dp[j - p[i]] + p[i]);
            }
        }
        if (dp[n] > 0)
            puts("YES");
        else
            puts("NO");
    }

    return 0;
}

https://codeforces.com/contest/1416/problem/A
Tag:思维 1500
思路：考虑数字i可以作为区间最小值出现的区间最小长度是多少，然后树状数组更新

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
const int N=3e5+10;
int t;
vector<int> v[N];
int f[N];
int lowbit(int x)
{
    return x&-x;
}
void add(int x,int val)
{
    while(x<N)
    f[x]=min(f[x],val),x+=lowbit(x);
}
int query(int x)
{
    int res=INF;
    while(x)
    {
        res=min(f[x],res);
        x-=lowbit(x);
    }
    return res;
}
int main()
{
    cin>>t;
    while(t--)
    {
        int n;cin>>n;for(int i=1;i<=n;i++) f[i]=INF;
        int mx=0;
        for(int i=1;i<=n;i++)
        {
            int x;cin>>x;v[x].push_back(i);
            mx=max(mx,x);
        }
        for(int i=1;i<=mx;i++)
        {
            if(!v[i].size()) continue;
            int len=max(v[i][0],n-v[i].back()+1);
            for(int j=1;j<v[i].size();j++)
            len=max(len,v[i][j]-v[i][j-1]);
            add(len,i);
        }
        for(int i=1;i<=n;i++)
        {
            int x=query(i);
            if(x==INF) cout<<-1<<" ";
            else cout<<x<<" ";
        }
        puts("");

        for(int i=1;i<=mx;i++) v[i].clear();
    }


    return 0;
}

https://codeforces.com/contest/1416/problem/B
Tag:构造，思维题 2000
由于a_i大于等于1，那么考虑前i-1的和一定大于等于i%a_i，所以可以先把所有的ai都转移到第一项上

#include<bits/stdc++.h>
using namespace std;
#define int long long
#define INF 0x3f3f3f3f
const int N=3e5+10;
int t;
int n,sum;
int f[N];
struct node{
    int a,b,c;
}st[N*3];
int cnt;
void solve()
{
    if(sum%n)
    {
        cout<<-1<<"\n";
        return ;
    }
    int p=sum/n;
    for(int i=2;i<=n;i++)
    {
        st[cnt++]={1,i,(i-f[i]%i)%i};
        f[i]+=(i-f[i]%i)%i;
        st[cnt++]={i,1,f[i]/i};
        
    }
    for(int i=2;i<=n;i++)
    {
        st[cnt++]={1,i,p};
    }

    cout<<cnt<<"\n";
    for(int i=0;i<cnt;i++) cout<<st[i].a<<" "<<st[i].b<<" "<<st[i].c<<"\n";
    return ;
}
signed main()
{
    int t;cin>>t;
    while(t--)
    {
        cin>>n;
        sum=0;
        cnt=0;
        for(int i=1;i<=n;i++) cin>>f[i],sum+=f[i];
        solve();
    }
    return 0;
}

https://codeforces.com/contest/1416/problem/C
Tag:01trie,贪心 2000
首先可以通过01trie求逆序对的数量，只需要看最高位不一样的就可以，然后考虑异或x的这一位是0和是1对最终答案的影响

#include <bits/stdc++.h>
using namespace std;
#define int long long
#define ll long long
const int N = 1e5 + 10;
int n, a[3*N], trie[N*40][2], p, ans;
int cnt;
int dp[50][2];
vector<int> v[N * 40];
inline ll read()
{
    int x = 0, f = 1;
    char c = getchar();
    while (c < '0' || c > '9')
    {
        if (c == '-')
            f = -1;
        c = getchar();
    }
    while (c >= '0' && c <= '9')
    {
        x = x * 10 + c - '0';
        c = getchar();
    }
    return f * x;
}
void insert(int x,int pos){
    int rt = 0;
    for(int i=30;i>=0;i--){
        int op = (x>>i&1)?1:0;
        if(!trie[rt][op]) trie[rt][op] = ++cnt;
        rt = trie[rt][op];
        v[rt].push_back(pos);
    }
}
void dfs(int pos, int u)
{
    if (pos == -1)
        return;
    int lm = trie[u][0], rm = trie[u][1];
    int sl = v[lm].size(), rl = v[rm].size();
    int c = 0;
    int r = 0;
    for (int i = 0; i < sl; i++)
    {
        while (r < rl && v[rm][r] < v[lm][i])
            r++;
        c += r;
    }
    dp[pos][0] += c;
    dp[pos][1] += sl * rl - c;
    if (lm)
        dfs(pos - 1, lm);
    if (rm)
        dfs(pos - 1, rm);
}
signed main()
{
    int t =1;
    while (t--)
    {
        n = read();
        for (int i = 0; i < n; i++)
        {
            a[i] = read();
            insert(a[i],  i);
        }
        dfs(30, 0);
        int ans = 0, res = 0;
        for (int i = 0; i <= 30; i++)
        {
            if (dp[i][0] <= dp[i][1])
                ans += dp[i][0];
            else
            {
                ans += dp[i][1];
                res += (1 << i);
            }
        }
        cout << ans << " " << res << " ";
    }
    return 0;
}
