土狗daisiki的日常坐牢

· 2022-02-07 ·


https://codeforces.com/contest/1508/problem/A
Tag:字符串，思维构造
题意：给三个由0和1构成的字符p_i，长度为2n，构造一个长度为3n的字符串S，让至少两个p是S的子序列
思路：首先可以得出任意p_i的cnt_0或者cnt_1是大于等于n的，考虑为什么是三个p?抽屉原理，一定会有两个p同时满足cnt1或者cnt0大于等于n，这样我们就固定好了n的长度，剩下的长度是2n,直接放就可以

#include <bits/stdc++.h>
using namespace std;
string s[3];
int f[3];
int t, n;
string check(int a, int b)
{
	char x;
	string ans;
	if (f[a] >= n)
		x = '1';
	else
		x = '0';
	int l = 0, r = 0;
	for (int i = 1; i <= n; i++)
	{
		while (l < 2 * n && s[a][l] != x)
			ans += s[a][l++];
		while (r < 2 * n && s[b][r] != x)
			ans += s[b][r++];
		l++, r++;
		ans += x;
	}
	while (l < 2 * n)
		ans += s[a][l++];
	while (r < 2 * n)
		ans += s[b][r++];
	return ans;
}
void solve()
{
	for (int i = 0; i < 3; i++)
	{
		for (int j = 0; j < 3; j++)
		{
			if (j == i)
				continue;
			if (f[i] > n && f[j] < n)
				continue;
			if (f[i] < n && f[j] > n)
				continue;
			string ss = check(i, j);
			if (ss.size() <= n * 3)
			{
				cout << ss << "\n";
				return;
			}
		}
	}
}
int main()
{
	cin >> t;
	while (t--)
	{
		cin >> n;
		f[0] = f[1] = f[2] = 0;
		for (int i = 0; i < 3; i++)
		{
			cin >> s[i];
			for (int j = 0; j < 2 * n; j++)
				if (s[i][j] == '1')
					f[i]++;
		}
		solve();
	}

	return 0;
}
