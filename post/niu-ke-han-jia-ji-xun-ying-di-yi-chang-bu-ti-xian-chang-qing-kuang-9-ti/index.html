牛客寒假集训营第一场补题(现场情况：9题)

· 2022-01-25 ·

B：在%3意义下一样的数，在同一区间内分数的变化情况是一样的，考虑倍增去维护

#include<bits/stdc++.h>
using namespace std;
const int N=2e5+10;
int n,q;
char s[N];
int st[3][N][21];
void init()
{
    for(int i=0;i<=20;i++)
    {
        for(int j=1;j<=n;j++)
        {
            if(i==0)
            {
                if(s[j]=='W') st[0][j][i]=st[1][j][i]=st[2][j][i]=1;
                if(s[j]=='L')
                {
                    st[1][j][i]=st[2][j][i]=-1;
                    st[0][j][i]=0;
                }
                if(s[j]=='D')
                st[0][j][i]=st[1][j][i]=st[2][j][i]=0;
            }
            else
            {
                int mid=j+(1<<(i-1));
                if(mid>n)
                {
                    for(int k=0;k<3;k++) 
                    st[k][j][i]=st[k][j][i-1];
                }
                else
                {
                    for(int k=0;k<3;k++)
                    st[k][j][i]=st[k][j][i-1]+st[((k+st[k][j][i-1])%3+3)%3][mid][i-1];
                }
            }
        }
    }
}
int main()
{
    scanf("%d%d",&n,&q);
    scanf("%s",s+1);
    init();
    while(q--)
    {
        int a,b,c; scanf("%d%d%d",&a,&b,&c);
        int l=a;
        while(l<=b)
        {
            int j=0;
            while(l+(1<<j)-1<=b) j++;
            j--;
            c+=st[c%3][l][j];
            l=l+(1<<j);
        }
        cout<<c<<"\n";
    }
    return 0;
}

#include <bits/stdc++.h>

int n;

    if (c[n] == 'G')

        f[n][1][1][1] = 3;
        f[n][1][1][2] = 2;
        f[n][1][2][1] = 2;
        f[n][2][1][1] = 2;
        f[n][1][1][3] = 2;
        f[n][1][3][1] = 2;
        f[n][3][1][1] = 2;
        f[n][1][3][3] = 1;
        f[n][3][1][3] = 1;
        f[n][3][3][1] = 1;
    }
    else if (c[n] == 'R')
    {
        f[n][2][2][2] = 0;
        f[n][2][2][3] = 0;
        f[n][2][3][2] = 0;
        f[n][3][2][2] = 0;
        f[n][2][3][3] = 0;
        f[n][3][2][3] = 0;
        f[n][3][3][2] = 0;
        f[n][1][2][2] = 1;
        f[n][2][1][2] = 1;
        f[n][2][2][1] = 1;
    }

    {
        f[n][3][3][3] = 0;

                if (f[i + 1][1][3][j] != -1)
                    f[i][1][1][3] = max(f[i][1][1][3], f[i + 1][1][3][j] + 1);
                if (f[i + 1][3][1][j] != -1)
                    f[i][1][3][1] = max(f[i][1][3][1], f[i + 1][3][1][j] + 1);
                if (f[i + 1][1][1][j] != -1)
                    f[i][3][1][1] = max(f[i][3][1][1], f[i + 1][1][1][j]);
                if (f[i + 1][1][2][j] != -1)
                    f[i][1][1][2] = max(f[i][1][1][2], f[i + 1][1][2][j] + 1);
                if (f[i + 1][2][1][j] != -1)
                    f[i][1][2][1] = max(f[i][1][2][1], f[i + 1][2][1][j] + 1);
                if (f[i + 1][1][1][j] != -1)
                    f[i][2][1][1] = max(f[i][2][1][1], f[i + 1][1][1][j]);
                if (f[i + 1][3][3][j] != -1)
                    f[i][1][3][3] = max(f[i][1][3][3], f[i + 1][3][3][j] + 1);
                if (f[i + 1][1][3][j] != -1)
                    f[i][3][1][3] = max(f[i][3][1][3], f[i + 1][1][3][j]);

                    f[i][3][3][1] = max(f[i][3][3][1], f[i + 1][3][1][j]);

        if (c[i] == 'R')

                    f[i][3][2][2] = max(f[i][3][2][2], f[i + 1][2][2][j] + 0);

                    f[i][1][2][3] = max(f[i][1][2][3], f[i + 1][2][3][j] + 1);

    int ans = -1;
    for (int i = 0; i <= 3; i++)
