A:
考虑对于二进制的某一位，如果1的数量大于0的数量那么这一位就需要是1

#include <bits/stdc++.h>
#define LOCAL
using namespace std;
int c1[50], c2[50];
int main()
{
#if ONLINE_JUDGE
#else
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--)
    {
        int n, k;
        cin >> n >> k;
        for (int i = 0; i < k; i++)
            c1[i] = c2[i] = 0;
        while (n--)
        {
            int x;
            cin >> x;
            for (int i = 0; i < k; i++)
            {
                if (x % 2 == 1)
                    c1[i]++;
                else
                    c2[i]++;
                x /= 2;
            }
        }
    
        int a = 1, res = 0;
        for (int i = 0; i < k; i++)
        {
            if (c1[i] > c2[i])
                res = res + a;
            a = a * 2;
        }
        cout << res << "\n";
    }

    return 0;
}

很容易想到体中所说的匹配指的是存在两个一样的k，他们前边有a个数，后边有b个数，让a+b+1最大
然后我们发现长度就是n-dis(k1,k2)

#include <bits/stdc++.h>
#define LOCAL
using namespace std;
const int N=2e5+10;
vector<int> e[N];
int a[N];
int main()
{
#if ONLINE_JUDGE
#else
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
#endif
    int t;
    cin >> t;
    while (t--)
    {
        for(int i=1;i<=150000;i++) e[i].clear();
        int n, k;
        cin >> n;
        for(int i=1;i<=n;i++)
        {
            int x;cin>>x;
            e[x].push_back(i);
        }
        int res=-1;

        for(int i=1;i<=150000;i++)
        {
            if(e[i].size()>=2)
            {
                k=0x3f3f3f3f;
                for(int j=1;j<e[i].size();j++)
                {
                    k=min(k,e[i][j]-e[i][j-1]);
                }
                res=max(res,n-k);
            }
        }
        cout<<res<<"\n";
    }

    return 0;
}

C：考虑第i个位置是从第j个位置转移来的，n3dp

#include <bits/stdc++.h>
using namespace std;
const int N = 550;
int a[N], d[N], dp[N][N];
int n, l, k;
int main()
{
	cin >> n >> l >> k;
	for (int i = 1; i <= n; i++)
		cin >> a[i];
	a[n + 1] = l;
	for (int i = 1; i <= n; i++)
		cin >> d[i];
	d[n + 1] = 0;
	n++;
	memset(dp, 0x3f, sizeof dp);
	dp[1][0] = 0;
	for (int i = 2; i <= n; i++){
		for (int j = 0; j <= k; j++){
			dp[i][j] = dp[i - 1][j] + (a[i] - a[i - 1]) * d[i - 1];
			for(int s=1;s<i;s++)
			{
				if(j-(i-s-1)<0) continue;
				dp[i][j]=min(dp[i][j],dp[s][j-(i-s-1)]+(a[i]-a[s])*d[s]); 
			}
		}
	}
	int res=0x3f3f3f3f;
	for(int i=0;i<=k;i++) res=min(res,dp[n][i]);
	cout<<res<<"\n";
	return 0;
}
