1.字符串的Border,如果字符串S的同长度的前缀与后缀完全相同，则称此前缀为一个border
例如：S=bbabbab,则他的border有b,bbab,字符串本身也可以是border，具体根据题意判断
2.周期与循环节
用∣\mid∣s∣\mid∣代表s的长度，下标从1开始
对于p，如果满足p < i ≤\leq≤ ∣\mid∣s∣\mid∣ 都有s(i)=s(i-p),那么p是s的一个周期
特别：∣\mid∣s∣\mid∣也是s的一个周期
循环节：若字符串的周期满足 p∣\mid∣ ∣\mid∣s∣\mid∣ ,则p是s的循环节
特别：∣\mid∣s∣\mid∣也是s的循环节
结论：
p是s的周期⟸\Longleftarrow⟸ ⟹\Longrightarrow⟹ ∣\mid∣s∣\mid∣-p是s的border
传递性S的border的border也是S的border
求S的所有border等价于求S的所有前缀的最大border
KMP
next数组：
next(i)=preffix(i)的非平凡的最大border(非平凡指去掉本身)
next(1)=0
考虑preffix(i)的大于1的border，去掉最后一个就变成了preffix(i-1)的border，所以求next(i)的时候，遍历preffix(i-1)的所有border，即next(i-1),next(next(i-1))....0,验证后一个字符是不是等于s(i)即可

#include <bits/stdc++.h>
#define x first
#define y second
using namespace std;
const int N = 1000010;
char a[N], b[N];
int nxt[N], n, m;

void kmp_nxt(char *s)
{
    for (int i = 2, j = 0; i <= m; ++i)
    {

        nxt[i] = nxt[i - 1];
        while (nxt[i] && s[i] != s[nxt[i] + 1])
            nxt[i] = nxt[nxt[i]];
        nxt[i] += (s[i] == s[nxt[i] + 1]);
    }
}

void solve() 
{
    scanf("%s%s", a + 1, b + 1);
    n = strlen(a + 1), m = strlen(b + 1);
    kmp_nxt(b);
    for (int i = 1, j = 0; i <= n; ++i)
    {
        while (j && a[i] != b[j + 1])
            j = nxt[j];
        if (a[i] == b[j + 1])
            ++j;
        if (j == m)
        {
            printf("%d\n", i - j + 1);
        }
    }

    for (int i = 1; i <= m; ++i)
        printf("%d ", nxt[i]);
}

int main()
{
    int _ = 1;
    while (_--)
        solve();
    return 0;
}

例题：
对于字符串S，求他的最长子串p，满足p是S的前缀，p是S的后缀，同时p至少在S中出现三次
思路：首先p是S的border，可以通过kmp求出，由于p要最长且出现三次，可以确定答案一定在nxt(n)与nxt(nxt(n))之间，因为nxt(nxt(n))在S中至少出现了四次，所以答案就在这两个里边，注意特判掉不存在的情况

#include<bits/stdc++.h>
using namespace std;
const int N=1e6+10;
char s[N];
int nxt[N];
int len;
void kmp()
{
	nxt[1]=0;
	for(int i=2;i<=len;i++)
	{
		nxt[i]=nxt[i-1];
		while(nxt[i]&&s[i]!=s[nxt[i]+1]) nxt[i]=nxt[nxt[i]];
		nxt[i]+=(s[i]==s[nxt[i]+1]);
	}
}
int main()
{

	len=strlen(s+1);
	kmp();

Border的性质：
周期定理：若p,q都是S的周期，那么gcd(p,q)，也是S的周期
一个串的Border数量是O(n)个，但是他们组成了(logN)个等差数列
Border树：
对于一个字符串S，他的border树(nxt树)共有n+1个节点
0是根节点，对于其他节点i的父节点是nxt(i)
性质：
每个前缀的所有border，节点i到根的链
哪些前缀有长度为x的border：x的子树
求两个前缀的公共border等于求LCA

Trie
字典：一个字符串的集合称为字典
字典串：字典中的串称为字典串
trie是一颗有根树，每个点最多有字符集大小个后继边，每个边上有一个字符，每个点表示一个前缀串，每个点还有一个终止标记
例题：https://ac.nowcoder.com/acm/problem/15049
考虑怎样的串可以成为最小字典序，首先字典中不能存在他的前缀，其次根据他的大小不能产生环，比如abcb,abcc，acb三个串，如果abcc想成为最小字典序，那么对于第二个字符需要有b小于c，对于第四个字符需要满足c小于b，有环，不符合要求
解法：构建trie，每个串跑拓扑排序

#include <bits/stdc++.h>
using namespace std;
const int N = 3e5 + 5;
int n, p = 1, trie[N][26], vis[N], deg[N];
string s[30005];
vector<string> ans;
vector<int> v[30];
void insert(string a, int len)
{
    int u = 1;
    for (int i = 0; i < len; i++)
    {
        int c = a[i] - 'a';
        if (!trie[u][c])
            trie[u][c] = ++p;
        u = trie[u][c];
    }
    vis[u] += 1;
}
bool topsort()
{
    queue<int> q;
    for (int i = 0; i < 26; i++)
        if (!deg[i])
            q.push(i);
    while (!q.empty())
    {
        auto f = q.front();
        q.pop();
        for (auto x : v[f])
        {
            deg[x]--;
            if (deg[x] == 0)
                q.push(x);
        }
    }
    for (int i = 0; i < 26; i++)
        if (deg[i] > 0)
            return false;
    return true;
}
bool check(string s, int len)
{
    int u = 1;
    for (int i = 0; i < 26; i++)
        v[i].clear(), deg[i] = 0;
    for (int i = 0; i < len; i++)
    {
        if (vis[u])
            return false;
        int x = s[i] - 'a';
        for (int j = 0; j < 26; j++)
        {
            if (trie[u][j] && j != x)
            {
                deg[j]++;
                v[x].push_back(j);
            }
        }
        u = trie[u][x];
    }
    return topsort();
}
int main()
{
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        cin >> s[i];
        int len = s[i].size();
        insert(s[i], len);
    }
    for (int i = 1; i <= n; i++)
    {
        int len = s[i].size();
        if (check(s[i], len))
            ans.emplace_back(s[i]);
    }
    cout << ans.size() << "\n";
    for (auto x : ans)
        cout << x << "\n";
    return 0;
}

01Trie，对于二进制的题可以考虑01Trie

#include <bits/stdc++.h>
using namespace std;
#define ll long long
int n, a[1000005], trie[10000005][2], cnt, ans;
inline ll read()
{
    int x = 0, f = 1;
    char c = getchar();
    while (c < '0' || c > '9')
    {
        if (c == '-')
            f = -1;
        c = getchar();
    }
    while (c >= '0' && c <= '9')
    {
        x = x * 10 + c - '0';
        c = getchar();
    }
    return f * x;
}
void insert(int x){
    int rt = 0;
    for(int i=30;i>=0;i--){
        int op = (x>>i&1)?1:0;
        if(!trie[rt][op]) trie[rt][op] = ++cnt;
        rt = trie[rt][op];
    }
}

int search(int u, int x)
{
    int res = 0, t = u;
    for (int i = 30; i >= 0; i--)
    {
        if (trie[t][!((x >> i) & 1)])
        {
            res += 1 << i;
            t = trie[t][!((x >> i) & 1)];
        }
        else
        {
            t = trie[t][(x >> i) & 1];
        }
    }
    return res;
}

int main()
{
    n = read();
    for (int i = 0; i < n; i++)
    {
        a[i] = read();
        insert(a[i]);
    }
    for (int i = 0; i < n; i++)
    {
        ans = max(ans, search(0, a[i]));
    }
    printf("%d\n", ans);
    return 0;
}
